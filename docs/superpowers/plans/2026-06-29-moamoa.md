# MoaMoa Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 생일선물 크라우드펀딩 서비스 MoaMoa를 구축한다 — 호스트가 펀딩 페이지를 생성하고 공유 링크로 참여자를 모으며, 토스페이먼츠(테스트 모드)로 결제하고 Supabase Realtime으로 현황을 실시간 표시한다.

**Architecture:** Server Actions가 service_role 키로 모든 DB 쓰기를 처리하고(RLS 우회), 단일 API Route(`/api/payment/confirm`)가 토스페이먼츠 서버 검증을 담당한다. Supabase Realtime이 `payments` 테이블 변경을 구독해 공개 펀딩 페이지의 진행 현황을 실시간 갱신한다.

**Tech Stack:** Next.js 14 (App Router), TypeScript, Supabase (PostgreSQL + Email OTP + Realtime), @supabase/ssr, TossPayments SDK (@tosspayments/payment-sdk), Kakao Share SDK, Tailwind CSS, Vitest, nanoid

## Global Constraints

- Node.js 18+, Next.js 14 App Router (Pages Router 사용 금지)
- TypeScript strict mode 활성화
- 모든 UI 텍스트는 한국어
- 최소 결제 금액: 1,000원
- share_token: 소문자 영숫자 8자 (nanoid customAlphabet)
- DB 쓰기는 모두 Server Action + SUPABASE_SERVICE_ROLE_KEY 사용
- 브라우저 Supabase 클라이언트는 NEXT_PUBLIC_SUPABASE_ANON_KEY (읽기 + Realtime만)
- 토스페이먼츠 테스트 모드 전용 (실제 결제 없음)
- 서버 시크릿 키(TOSS_SECRET_KEY, SUPABASE_SERVICE_ROLE_KEY)는 서버 전용 — NEXT_PUBLIC_ 접두사 금지

```sql
create table fundings (
  id uuid primary key default gen_random_uuid(),
  creator_email text not null,
  title text not null,
  description text,
  end_date timestamptz not null,
  share_token text unique not null,

  created_at timestamptz not null default now()
);

-- gifts 테이블
create table gifts (
  id uuid primary key default gen_random_uuid(),
  funding_id uuid not null references fundings(id) on delete cascade,
  name text not null,
  target_amount integer not null check (target_amount > 0),
  description text,
  image_url text,
  created_at timestamptz not null default now()
);

-- payments 테이블
create table payments (
  id uuid primary key default gen_random_uuid(),
  funding_id uuid not null references fundings(id),
  participant_name text not null,
  message text,
  amount integer not null check (amount >= 1000),
  order_id text unique not null,
  payment_key text,
  status text not null default 'pending' check (status in ('pending', 'confirmed', 'failed')),
  created_at timestamptz not null default now()
);

-- RLS 활성화
alter table fundings enable row level security;
alter table gifts enable row level security;
alter table payments enable row level security;

-- fundings: anon 읽기 허용
create policy "fundings_select_all" on fundings for select using (true);

-- gifts: anon 읽기 허용
create policy "gifts_select_all" on gifts for select using (true);

-- payments: anon 읽기/쓰기 허용 (쓰기는 서버에서 처리하지만 Realtime 구독을 위해 select 필요)
create policy "payments_select_all" on payments for select using (true);
create policy "payments_insert_all" on payments for insert with check (true);
```

```typescript
// lib/supabase/server.ts
import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'
import type { Database } from './types'

export const createServerSupabaseClient = () => {
  const cookieStore = cookies()
  return createServerClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll: () => cookieStore.getAll(),
        setAll: (cookiesToSet) => {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            )
          } catch {}
        },
      },
    }
  )
}

export const createServiceRoleClient = () =>
  createServerClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    { cookies: { getAll: () => [], setAll: () => {} } }
  )
```

```tsx
type ButtonProps = React.ButtonHTMLAttributes<HTMLButtonElement> & {
  variant?: 'primary' | 'outline'
}

export default function Button({ variant = 'primary', className = '', children, ...props }: ButtonProps) {
  const base = 'w-full py-3 px-4 rounded-xl font-semibold text-sm transition-colors disabled:opacity-50 disabled:cursor-not-allowed'
  const variants = {
    primary: 'bg-rose-500 text-white hover:bg-rose-600',
    outline: 'border-2 border-rose-500 text-rose-500 hover:bg-rose-50',
  }
  return (
    <button className={`${base} ${variants[variant]} ${className}`} {...props}>
      {children}
    </button>
  )
}
```

```typescript
// app/create/actions.ts
'use server'

import { createServerSupabaseClient, createServiceRoleClient } from '@/lib/supabase/server'
import { generateShareToken } from '@/lib/utils'

export async function sendOtp(email: string) {
  const supabase = createServerSupabaseClient()
  const { error } = await supabase.auth.signInWithOtp({
    email,
    options: { shouldCreateUser: true },
  })
  if (error) return { error: error.message }
  return { success: true as const }
}

export async function verifyOtp(email: string, otp: string) {
  const supabase = createServerSupabaseClient()
  const { error } = await supabase.auth.verifyOtp({
    email,
    token: otp,
    type: 'email',
  })
  if (error) return { error: 'OTP가 올바르지 않거나 만료되었습니다' }
  return { success: true as const }
}

type GiftInput = { name: string; targetAmount: number; description: string }

export async function createFunding(data: {
  email: string
  title: string
  description: string
  endDate: string
  gifts: GiftInput[]
}) {
  if (data.gifts.length === 0) return { error: '선물을 1개 이상 추가해주세요' }

  const supabase = createServiceRoleClient()
  const shareToken = generateShareToken()

  const { data: funding, error: fundingError } = await supabase
    .from('fundings')
    .insert({
      creator_email: data.email,
      title: data.title,
      description: data.description,
      end_date: new Date(data.endDate).toISOString(),
      share_token: shareToken,
      status: 'active',
    })
    .select()
    .single()

  if (fundingError) return { error: fundingError.message }

  const { error: giftsError } = await supabase.from('gifts').insert(
    data.gifts.map((g) => ({
      funding_id: funding.id,
      name: g.name,
      target_amount: g.targetAmount,
      description: g.description,
    }))
  )

  if (giftsError) return { error: giftsError.message }

  return { shareToken }
}
```

```tsx
// app/funding/[token]/FundingRealtime.tsx
'use client'

import { useEffect, useState } from 'react'
import { createClient } from '@/lib/supabase/client'
import FundingProgress from '@/components/funding/FundingProgress'
import GiftList from '@/components/funding/GiftList'
import DonorList from '@/components/funding/DonorList'
import type { Gift, Payment } from '@/lib/supabase/types'

type Props = {
  fundingId: string
  initialPayments: Payment[]
  gifts: Gift[]
  totalTarget: number
  endDate: string
}

export default function FundingRealtime({ fundingId, initialPayments, gifts, totalTarget, endDate }: Props) {
  const [payments, setPayments] = useState<Payment[]>(initialPayments)
  const supabase = createClient()

  useEffect(() => {
    const channel = supabase
      .channel(`funding-${fundingId}`)
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'payments',
          filter: `funding_id=eq.${fundingId}`,
        },
        (payload) => {
          if (payload.eventType === 'INSERT') {
            const newPayment = payload.new as Payment
            if (newPayment.status === 'confirmed') {
              setPayments((prev) => [newPayment, ...prev])
            }
          }
          if (payload.eventType === 'UPDATE') {
            const updated = payload.new as Payment
            setPayments((prev) =>
              prev
                .map((p) => (p.id === updated.id ? updated : p))
                .filter((p) => p.status === 'confirmed')
            )
            // confirmed로 바뀐 새 항목 추가
            if (updated.status === 'confirmed') {
              setPayments((prev) => {
                const exists = prev.find((p) => p.id === updated.id)
                return exists ? prev.map((p) => (p.id === updated.id ? updated : p)) : [updated, ...prev]
              })
            }
          }
        }
      )
      .subscribe()

    return () => { supabase.removeChannel(channel) }
  }, [fundingId, supabase])

  const confirmedPayments = payments.filter((p) => p.status === 'confirmed')
  const totalAmount = confirmedPayments.reduce((sum, p) => sum + p.amount, 0)

  return (
```

```typescript
'use server'

import { createServiceRoleClient } from '@/lib/supabase/server'

export async function createPendingPayment(
  fundingId: string,
  participantName: string,
  message: string,
  amount: number
) {
  if (amount < 1000) return { error: '최소 결제 금액은 1,000원입니다' }
  if (!participantName.trim()) return { error: '이름을 입력해주세요' }

  const orderId = `moamoa-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`

  const { error } = await supabase.from('payments').insert({
    funding_id: fundingId,
    participant_name: participantName.trim(),
    message: message.trim() || null,
    amount,
    order_id: orderId,
    status: 'pending',
  })

  if (error) return { error: error.message }
  return { orderId }
}
```

```typescript
// app/api/payment/confirm/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { createServiceRoleClient } from '@/lib/supabase/server'

export async function POST(req: NextRequest) {
  const body = await req.json().catch(() => null)
  if (!body) return NextResponse.json({ error: '잘못된 요청입니다' }, { status: 400 })

  const { paymentKey, orderId, amount } = body
  if (!paymentKey || !orderId || !amount) {
    return NextResponse.json({ error: '필수 파라미터 누락' }, { status: 400 })
  }

  const supabase = createServiceRoleClient()

  // pending 결제 조회
  const { data: payment, error: findError } = await supabase
    .from('payments')
    .select('*')
    .eq('order_id', orderId)
    .eq('status', 'pending')
    .single()

  if (findError || !payment) {
    return NextResponse.json({ error: '결제 정보를 찾을 수 없습니다' }, { status: 404 })
  }

  // 금액 일치 검증
  if (payment.amount !== Number(amount)) {
    return NextResponse.json({ error: '결제 금액이 일치하지 않습니다' }, { status: 400 })
  }

  // 토스페이먼츠 서버 검증
  const tossResponse = await fetch('https://api.tosspayments.com/v1/payments/confirm', {
    method: 'POST',
    headers: {
      Authorization: `Basic ${Buffer.from(`${process.env.TOSS_SECRET_KEY}:`).toString('base64')}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({ paymentKey, orderId, amount: Number(amount) }),
  })

  if (!tossResponse.ok) {
    const tossError = await tossResponse.json().catch(() => ({}))
    await supabase.from('payments').update({ status: 'failed' }).eq('order_id', orderId)
    return NextResponse.json(
      { error: tossError.message || '토스페이먼츠 확인 실패' },
      { status: 400 }
    )
  }

  // DB 업데이트
  await supabase
    .from('payments')
    .update({ status: 'confirmed', payment_key: paymentKey })
    .eq('order_id', orderId)

  return NextResponse.json({ success: true })
}
```

```typescript
// app/funding/[token]/admin/actions.ts
'use server'

import { createServerSupabaseClient, createServiceRoleClient } from '@/lib/supabase/server'

export async function sendAdminOtp(token: string, email: string) {
  const supabase = createServiceRoleClient()

  // 이메일이 creator_email과 일치하는지 먼저 확인
  const { data: funding } = await supabase
    .from('fundings')
    .select('creator_email')
    .eq('share_token', token)
    .single()

  if (!funding) return { error: '펀딩을 찾을 수 없습니다' }
  if (funding.creator_email !== email) return { error: '주최자 이메일이 일치하지 않습니다' }

  const authSupabase = createServerSupabaseClient()
  const { error } = await authSupabase.auth.signInWithOtp({
    email,
    options: { shouldCreateUser: true },
  })

  if (error) return { error: error.message }
  return { success: true as const }
}

export async function verifyAdminOtp(email: string, otp: string) {
  const supabase = createServerSupabaseClient()
  const { error } = await supabase.auth.verifyOtp({
    email,
    token: otp,
    type: 'email',
  })
  if (error) return { error: 'OTP가 올바르지 않거나 만료되었습니다' }
  return { success: true as const }
}

export async function requestSettlement(token: string, creatorEmail: string) {
  const supabase = createServiceRoleClient()

  // 이메일 재검증
  const { data: funding } = await supabase
    .from('fundings')
    .select('*')
    .eq('share_token', token)
    .single()

  if (!funding) return { error: '펀딩을 찾을 수 없습니다' }
  if (funding.creator_email !== creatorEmail) return { error: '권한이 없습니다' }
  if (funding.status === 'closed') return { error: '이미 정산된 펀딩입니다' }

  // 테스트 모드: 실제 정산 API 호출 대신 상태만 변경
  const { error } = await supabase
    .from('fundings')
    .update({ status: 'closed' })
    .eq('share_token', token)

  if (error) return { error: error.message }
  return { success: true as const }
}
```

- [ ] **Step 2: app/funding/[token]/admin/page.tsx 작성**

```tsx
// app/funding/[token]/admin/page.tsx
'use client'

import { useState, useTransition } from 'react'
import { useParams } from 'next/navigation'
import { createClient } from '@/lib/supabase/client'
import Button from '@/components/ui/Button'
import Input from '@/components/ui/Input'
import { formatKRW } from '@/lib/utils'
import { sendAdminOtp, verifyAdminOtp, requestSettlement } from './actions'
import type { Payment } from '@/lib/supabase/types'

type AdminData = {
  title: string
  totalAmount: number
  payments: Payment[]
  status: string
}

export default function AdminPage() {
  const params = useParams<{ token: string }>()
  const [isPending, startTransition] = useTransition()
  const [step, setStep] = useState<'email' | 'otp' | 'admin'>('email')
  const [email, setEmail] = useState('')
  const [otp, setOtp] = useState('')
  const [error, setError] = useState('')
  const [adminData, setAdminData] = useState<AdminData | null>(null)
  const [settled, setSettled] = useState(false)

  const handleSendOtp = () => {
    setError('')
    startTransition(async () => {
      const result = await sendAdminOtp(params.token, email)
      if ('error' in result) return setError(result.error)
      setStep('otp')
    })
  }

  const handleVerifyOtp = () => {
    setError('')
    startTransition(async () => {
      const result = await verifyAdminOtp(email, otp)
      if ('error' in result) return setError(result.error)

      // 관리자 데이터 로드
      const supabase = createClient()
      const { data: funding } = await supabase
        .from('fundings')
        .select('title, status')
        .eq('share_token', params.token)
        .single()

      const { data: payments } = await supabase
        .from('payments')
        .select('*')
        .eq('status', 'confirmed')
        .order('created_at', { ascending: false })

      if (funding) {
        const total = (payments ?? []).reduce((s, p) => s + p.amount, 0)
        setAdminData({
          title: funding.title,
          status: funding.status,
          totalAmount: total,
          payments: payments ?? [],
        })
        setSettled(funding.status === 'closed')
      }
      setStep('admin')
    })
  }

  const handleSettle = () => {
    startTransition(async () => {
      const result = await requestSettlement(params.token, email)
      if ('error' in result) return setError(result.error)
      setSettled(true)
    })
  }

  return (
    <main className="min-h-screen flex items-center justify-center px-4">
      <div className="w-full max-w-md flex flex-col gap-5">
        {step === 'email' && (
          <>
            <h1 className="text-xl font-bold">주최자 확인</h1>
            <p className="text-sm text-gray-500">펀딩 생성 시 입력한 이메일을 입력해주세요</p>
            <Input label="이메일" type="email" value={email} onChange={(e) => setEmail(e.target.value)} error={error} />
            <Button onClick={handleSendOtp} disabled={isPending}>인증 코드 받기</Button>
          </>
        )}

        {step === 'otp' && (
          <>
            <h1 className="text-xl font-bold">인증 코드 확인</h1>
            <p className="text-sm text-gray-500">{email}로 전송된 코드를 입력해주세요</p>
            <Input label="인증 코드" value={otp} onChange={(e) => setOtp(e.target.value)} maxLength={6} error={error} />
            <Button onClick={handleVerifyOtp} disabled={isPending}>확인</Button>
          </>
        )}

        {step === 'admin' && adminData && (
          <>
            <h1 className="text-xl font-bold">관리자 페이지</h1>
            <p className="text-lg font-semibold text-gray-700">{adminData.title}</p>

            <div className="bg-rose-50 rounded-2xl p-4">
              <p className="text-sm text-gray-500">총 모인 금액</p>
              <p className="text-2xl font-bold text-rose-500">{formatKRW(adminData.totalAmount)}</p>
            </div>

            <div className="flex flex-col gap-2">
              <h2 className="text-sm font-semibold text-gray-700">결제 내역 ({adminData.payments.length}건)</h2>
              {adminData.payments.map((p) => (
                <div key={p.id} className="bg-white border border-gray-100 rounded-xl p-3 flex justify-between text-sm">
                  <div>
                    <p className="font-semibold">{p.participant_name}</p>
                    {p.message && <p className="text-gray-400 text-xs">"{p.message}"</p>}
                  </div>
                  <span className="text-rose-500 font-semibold">{formatKRW(p.amount)}</span>
                </div>
              ))}
            </div>

            {error && <p className="text-sm text-red-500">{error}</p>}

            {settled ? (
              <div className="bg-green-50 border border-green-200 rounded-xl p-4 text-center text-green-700 text-sm font-semibold">
                ✅ 정산이 완료되었습니다
              </div>
            ) : (
              <Button onClick={handleSettle} disabled={isPending}>
                {isPending ? '정산 중...' : '정산 요청하기'}
              </Button>
            )}
          </>
        )}
      </div>
    </main>
  )
}
```
